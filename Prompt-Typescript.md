function 句末换行处理(文本: string): string {
  return 文本
    .replace('。', '。\n\n')
    .trim()
}

function 自动换行处理(文本: string, 最大字符数: number): string {
  const 字符数 = 文本.length
  if (字符数 > 最大字符数) {
    return 文本
      .split(最大字符数)
      .join('\n')
  }
  return 文本
}
